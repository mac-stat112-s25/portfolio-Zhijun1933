Bivariate Viz

# Scatter plots (continuous x continuous)
ggplot(mpg, aes(x = displ, y = hwy)) +
  geom_point()

ggplot(economics, aes(x = date, y = unemploy)) +
  geom_line()

# Boxplots (categorical x continuous)

  geom_boxplot()

# Bar plots (categorical x continuous, summarized)

Multivariate Viz

# Color for third variable
ggplot(mpg, aes(x = displ, y = hwy, color = class)) +
  geom_point()

  geom_point()

  geom_point()

# Faceting (small multiples)
ggplot(mpg, aes(x = displ, y = hwy)) +
  geom_point() +
  facet_wrap(~class)

Effective Viz

# Complete plot with proper annotations
ggplot(mpg, aes(x = displ, y = hwy, color = class)) +
  geom_point(alpha = 0.7) +
  labs(
    title = "Engine Displacement vs. Highway MPG",
    subtitle = "Larger engines tend to have lower fuel economy",
    x = "Engine Displacement (L)",
    y = "Highway Fuel Economy (MPG)",
    color = "Vehicle Class"
  ) +
  theme_minimal() +
  scale_color_brewer(palette = "Set1")

# Custom themes
ggplot(mpg, aes(x = displ, y = hwy)) +
  geom_point() +
  theme_bw() +      # Black and white theme
  theme(
    axis.title = element_text(face = "bold"),
    legend.position = "bottom"
  )

Wrangling

# Core dplyr verbs
# filter: subset rows
diamonds %>% filter(cut == "Ideal" & price > 1000)

# select: subset columns
diamonds %>% select(carat, cut, price)

# arrange: reorder rows
diamonds %>% arrange(desc(price))

# mutate: create/modify columns
diamonds %>% mutate(price_per_carat = price / carat)

# summarize: collapse rows
diamonds %>% summarize(avg_price = mean(price), 
                      count = n())

# Group operations
diamonds %>%
  group_by(cut) %>%
  summarize(avg_price = mean(price), 
            count = n()) %>%
  arrange(desc(avg_price))

# Convert wide to long format
stocks <- data.frame(
  time = as.Date('2009-01-01') + 0:9,
  X = rnorm(10, 0, 1),
  Y = rnorm(10, 0, 2),
  Z = rnorm(10, 0, 4)
)

stocks_long <- stocks %>%
  pivot_longer(cols = c(X, Y, Z),
               names_to = "stock",
               values_to = "price")

# Convert long to wide format
stocks_wide <- stocks_long %>%
  pivot_wider(names_from = stock,
              values_from = price)

                  last = c("Smith", "Johnson"))

Joining

# Sample data frames
employees <- data.frame(
  id = 1:5,
  name = c("Alice", "Bob", "Charlie", "David", "Eve")
)

salaries <- data.frame(
  id = c(1, 3, 4, 6),
  salary = c(50000, 60000, 55000, 65000)
)

# Inner join (only matching rows)
inner_join(employees, salaries, by = "id")

# Left join (all rows from employees)
left_join(employees, salaries, by = "id")

# Right join (all rows from salaries)
right_join(employees, salaries, by = "id")

# Full join (all rows from both)
full_join(employees, salaries, by = "id")

# Anti join (rows in employees with no match in salaries)
anti_join(employees, salaries, by = "id")

Factors

# Create factors
months <- factor(c("Jan", "Feb", "Mar"), 
                levels = c("Jan", "Feb", "Mar", "Apr"))
months

# Using forcats package for factor manipulation
library(forcats)

# Reorder factor levels
mpg %>%
  mutate(class = fct_reorder(class, hwy, .fun = median)) %>%
  ggplot(aes(x = class, y = hwy)) +
  geom_boxplot()

# Combine rare levels
diamonds %>%
  mutate(clarity = fct_lump_n(clarity, n = 5)) %>%
  count(clarity)
